
# ATALHOS TERMINAL #

pwd 
(mostra diretório atual)

ls
(lista conteúdo da pasta)

ls -la
(lista conteúdo da pasta incluindo arquivos ocultos)

cd 
(abre diretório)

mkdir 
(cria diretório)

mv 
(move arquivos/diretórios)

rm 
(remover)

clear 
(limpar terminal)

rs 
restart servidor)

code . 
(abre pelo terminal pasta desejada)
